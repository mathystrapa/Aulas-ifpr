No MySQL Workbench, há uma funcionalidade chamada "engenharia reversa" que pega um código SQL e transforma em um diagrama de classes.

Código do SQL:

----------------------------------------------------------------------------------------

CREATE TABLE PESSOAS(
CPF INT PRIMARY KEY,
NOME VARCHAR(200),
DATA_NASC DATE,
SEXO CHAR(1)
);

CREATE TABLE FUNCIONARIOS(
COD INT AUTO_INCREMENT PRIMARY KEY,
CPF INT,
SALARIO FLOAT,
CARGO VARCHAR(100),
FOREIGN KEY (CPF) REFERENCES PESSOAS(CPF)
);

CREATE TABLE ALUNOS(
MATRICULA INT AUTO_INCREMENT PRIMARY KEY,
TURMA VARCHAR(20),
CPF INT,
CPF_RESPONSAVEL INT,
FOREIGN KEY (CPF) REFERENCES PESSOAS(CPF),
FOREIGN KEY (CPF_RESPONSAVEL) REFERENCES PESSOAS(CPF)
);

CREATE TABLE PRODUTOS(
COD INT AUTO_INCREMENT PRIMARY KEY,
NOME VARCHAR(200),
PRECO FLOAT,
CUSTO FLOAT,
DESCRICAO VARCHAR(3000)
);

CREATE TABLE VEICULOS(
COD INT AUTO_INCREMENT PRIMARY KEY,
VIN INT,
MARCA VARCHAR(200),
MODELO VARCHAR(200),
ANO INT,
PLACA VARCHAR(7)
);

CREATE TABLE CONTAS_A_PAGAR(
COD INT AUTO_INCREMENT PRIMARY KEY,
NOME VARCHAR(100),
VALOR FLOAT,
CPF_PAGADOR INT,
CPF_DESTINO INT,
FOREIGN KEY (CPF_PAGADOR) REFERENCES PESSOAS(CPF),
FOREIGN KEY (CPF_DESTINO) REFERENCES PESSOAS(CPF)
);

CREATE TABLE COMPUTADOR(
COD INT AUTO_INCREMENT PRIMARY KEY,
COD_PRODUTO INT,
MARCA VARCHAR(100),
PROCESSADOR VARCHAR(100),
HD FLOAT,
RAM FLOAT,
FOREIGN KEY (COD_PRODUTO) REFERENCES PRODUTOS(COD)
);

CREATE TABLE SMARTPHONES(
COD INT AUTO_INCREMENT PRIMARY KEY,
COD_PRODUTO INT,
MARCA VARCHAR(50),
MODELO VARCHAR(50),
HD FLOAT,
RAM FLOAT,
FOREIGN KEY (COD_PRODUTO) REFERENCES PRODUTOS(COD)
);

CREATE TABLE ELETRODOMESTICOS(
COD INT AUTO_INCREMENT PRIMARY KEY,
COD_PRODUTO INT,
NOME VARCHAR(100),
PRECO FLOAT,
FOREIGN KEY (COD_PRODUTO) REFERENCES PRODUTOS(COD)
);

CREATE TABLE PAGAMENTO(
COD INT AUTO_INCREMENT PRIMARY KEY,
NOME VARCHAR(100),
VALOR FLOAT,
CPF_PAGADOR INT,
CPF_DESTINO INT,
FOREIGN KEY (CPF_PAGADOR) REFERENCES PESSOAS(CPF),
FOREIGN KEY (CPF_DESTINO) REFERENCES PESSOAS(CPF)
);
